package com.example.myapplication3;

import android.graphics.Bitmap;
import android.graphics.BitmapFactory;
import android.graphics.Canvas;
import android.graphics.Paint;
import android.graphics.Path;
import android.os.Bundle;
import android.view.View;
import android.widget.ImageView;

import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        ImageView mImageView = findViewById(R.id.triangleView);
        Bitmap original = BitmapFactory.decodeResource(getResources(), R.drawable.profile_image);
        Bitmap mask = BitmapFactory.decodeResource(getResources(), R.drawable.triangle_mask);

        // Create a new bitmap to hold the result
        Bitmap result = Bitmap.createBitmap(mask.getWidth(), mask.getHeight(), Bitmap.Config.ARGB_8888);

        // Create a canvas with the result bitmap
        Canvas mCanvas = new Canvas(result);

        // Paint for masking
        Paint paint = new Paint(Paint.ANTI_ALIAS_FLAG);
        paint.setXfermode(new PorterDuffXfermode(PorterDuff.Mode.DST_IN));

        // Draw the original image
        mCanvas.drawBitmap(original, 0, 0, null);

        // Draw the mask
        mCanvas.drawBitmap(mask, 0, 0, paint);

        // Clear the xfer mode
        paint.setXfermode(null);

        // Set the result bitmap to the ImageView
        mImageView.setImageBitmap(result);

        // Set the scale type
        mImageView.setScaleType(ImageView.ScaleType.CENTER);
    }
}
