        // Set up the path for the triangle to match the dimensions
        Path trianglePath = new Path();
        trianglePath.moveTo(triangleWidth / 2, 0); // top center
        trianglePath.lineTo(triangleWidth, triangleHeight); // bottom right
        trianglePath.lineTo(0, triangleHeight); // bottom left
        trianglePath.close();

        // Clip the canvas with the triangle path
        canvas.save();
        canvas.clipPath(trianglePath);
