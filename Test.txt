import android.content.Context;
import android.graphics.Canvas;
import android.graphics.Paint;
import android.graphics.Path;
import android.util.AttributeSet;
import androidx.appcompat.widget.AppCompatImageView;

public class TriangleOutlineView extends AppCompatImageView {

    private Paint paint;
    private Path path;

    public TriangleOutlineView(Context context) {
        super(context);
        init();
    }

    public TriangleOutlineView(Context context, AttributeSet attrs) {
        super(context, attrs);
        init();
    }

    public TriangleOutlineView(Context context, AttributeSet attrs, int defStyleAttr) {
        super(context, attrs, defStyleAttr);
        init();
    }

    private void init() {
        paint = new Paint();
        paint.setAntiAlias(true);
        paint.setStyle(Paint.Style.STROKE);
        paint.setStrokeWidth(5); // Set the stroke width as needed

        path = new Path();
        // Define your triangle outline path here
        path.moveTo(100, 100);
        path.lineTo(300, 100);
        path.lineTo(200, 300);
        path.close();
    }

    @Override
    protected void onDraw(Canvas canvas) {
        super.onDraw(canvas);
        canvas.drawPath(path, paint);
    }
}
